class SmaCrossover(bt.Strategy):
    params = (
        ('fast', 50),
        ('slow', 200),
    )

    def next(self):
        if not self.position:  # Not in the market
            if self.crossover > 0:  # Fast SMA crosses above Slow SMA
                self.buy()
        else:  # In the market
